DROP DATABASE IF EXISTS airport_db;
CREATE DATABASE airport_db;
USE airport_db;

CREATE TABLE passengers (
passenger_id INT AUTO_INCREMENT PRIMARY KEY,
first_name VARCHAR(50) NOT NULL,
last_name VARCHAR(50) NOT NULL,
email VARCHAR(100) UNIQUE NOT NULL
);

CREATE TABLE aircrafts (
aircraft_id INT AUTO_INCREMENT PRIMARY KEY,
model VARCHAR(50) NOT NULL,
capacity INT NOT NULL
);

CREATE TABLE flights (
flight_id INT AUTO_INCREMENT PRIMARY KEY,
flight_number VARCHAR(20) UNIQUE NOT NULL,
origin VARCHAR(50),
destination VARCHAR(50),
aircraft_id INT,
FOREIGN KEY (aircraft_id) REFERENCES aircrafts(aircraft_id)
);

CREATE TABLE tickets (
ticket_id INT AUTO_INCREMENT PRIMARY KEY,
passenger_id INT,
flight_id INT,
seat_number VARCHAR(10),
FOREIGN KEY (passenger_id) REFERENCES passengers(passenger_id),
FOREIGN KEY (flight_id) REFERENCES flights(flight_id)
);

-- Query 1: Passenger + Flight + Aircraft мэдээлэл
SELECT p.first_name, p.last_name, f.flight_number, a.model
FROM tickets t
JOIN passengers p ON t.passenger_id = p.passenger_id
JOIN flights f ON t.flight_id = f.flight_id
JOIN aircrafts a ON f.aircraft_id = a.aircraft_id;

-- Query 2: Нислэг бүрийн зорчигчийн тоо
SELECT f.flight_number, COUNT(t.passenger_id) AS total_passengers
FROM flights f
LEFT JOIN tickets t ON f.flight_id = t.flight_id
GROUP BY f.flight_number;

-- Query 3: Хамгийн их зорчигчтой нислэг
SELECT f.flight_number, COUNT(*) AS total
FROM tickets t
JOIN flights f ON t.flight_id = f.flight_id
GROUP BY f.flight_number
ORDER BY total DESC
LIMIT 1;

-- Query 4: 2 ба түүнээс дээш удаа ниссэн зорчигч
SELECT p.first_name, p.last_name, COUNT(*) AS flights_count
FROM tickets t
JOIN passengers p ON t.passenger_id = p.passenger_id
GROUP BY p.passenger_id, p.first_name, p.last_name
HAVING COUNT(*) >= 2;
